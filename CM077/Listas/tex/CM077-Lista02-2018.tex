\begin{enumerate}

\item Prove que o comprimento de qualquer curva ligando $p$ e $q$ no $\mathbb{R}^n$ \'e maior ou 
igual a $|p-q|$.

\item Prove que uma curva plana com curvatura constante est\'a contida em uma reta ou em um c\'irculo.

\item Sejam $\alpha,\beta\colon I\to \mathbb{R}^n$ duas curvas diferenci\'aveis
satisfazendo 
\begin{itemize}
	\item[(i)] $\alpha' = a\alpha + b\beta$.
	\item[(ii)] $\beta'= c\alpha + d\beta$.
\end{itemize}
Prove que se $a=-d$ ent\~ao $\alpha$ e $\beta$ pertencem a um mesmo plano.

\item Seja $\alpha\colon I\to\mathbb{R}^3$, uma curva no espaço com $|\alpha''|> 0$
	(n\~ao necessariamente p.p.c.a.). Prove que:
	\begin{itemize}
		\item[(i)] a curvatura em $\alpha(t)$ \'e $$k(t)=\frac{|\alpha'\wedge\alpha''|}{|\alpha'|^3};$$
		\item[(ii)] a tor\c c\~ao em $\alpha(t)$ \'e
			$$\tau(t)=-\frac{\langle\alpha'\wedge\alpha'',\alpha'''\rangle}{|\alpha'\wedge\alpha''|^2}.$$
	\end{itemize}
	
\end{enumerate}
